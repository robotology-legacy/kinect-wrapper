163  bool updateModule()
164  {
165  client.getDepthAndPlayers(depth,players);
166  client.getRgb(rgb);
167 
168  /*Alternatively you can ask for getJoints(joints), retrieving a deque of Players,
169  having information on all the players instead of having only information on
170  the closest one*/
171  bool tracked=client.getJoints(joint,KINECT_TAGS_CLOSEST_PLAYER);
172 
173  if (tracked)
174  client.getSkeletonImage(joint,skeletonImage);
175 
176  client.getPlayersImage(players,playersImage);
177  client.getDepthImage(depth,depthToDisplay);
178 
179  if (depthPort.getOutputCount()>0)
180  {
181  depthPort.prepare()=depthToDisplay;
182  depthPort.write();
183  }
184 
185  if (imagePort.getOutputCount()>0)
186  {
187  imagePort.prepare()=rgb;
188  imagePort.write();
189  }
190 
191  if (playersPort.getOutputCount()>0)
192  {
193  playersPort.prepare()=playersImage;
194  playersPort.write();
195  }
196 
197  if (skeletonPort.getOutputCount()>0)
198  {
199  skeletonPort.prepare()=skeletonImage;
200  skeletonPort.write();
201  }
202 
203  int u=160;
204  int v=120;
205  yarp::sig::Vector point3D;
206  client.get3DPoint(u,v,point3D);
207 
208  fprintf(stdout, "%s\n", point3D.toString().c_str());
209 
210  if (showImages)
211  {
212  cvConvertScale((IplImage*)depthToDisplay.getIplImage(),depthTmp,1.0/255);
213  cvShowImage("depth",depthTmp);
214  cvShowImage("players",(IplImage*)playersImage.getIplImage());
215  cvShowImage("skeleton",(IplImage*)skeletonImage.getIplImage());
216 
217  cvWaitKey(1);
218 
219  cvCvtColor((IplImage*)rgb.getIplImage(),rgbTmp,CV_BGR2RGB);
220  cvShowImage("rgb",rgbTmp);
221  }
222 
223  return true;
224  }
